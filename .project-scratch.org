* How to use - the TODO scanner
To use the TODO scanner, you can either press =C-c C-C= first in the codeblock under [[#config][Config]] to set up the todo scanning, then run the [[#scanner][TODO Scanner]]. The items will show up under the [[#todo_items][TODO Items]] heading. Haven't figured out how to stop the scanner from scanning this file...
Alternatively, you can just run =M-x org-babel-execute-buffer= (=C-c C-v b=) and all of the codeblocks will run, having your items appear under the [[#todo_items][TODO Items]] heading.

* 🔎 TODO Scanner
  :PROPERTIES:
  :CUSTOM_ID: scanner
  :VISIBILITY: folded
  :END:

  Run this block to scan the project and regenerate the *Pending* section above.
  Use =C-c C-c= with point inside the block.

  #+NAME: scan-todos
  #+BEGIN_SRC emacs-lisp :results none
  (require 'cl-lib)

  ;; ── helpers ────────────────────────────────────────────────────────────────

  (defun project-todo--excluded-p (path exclude-dirs)
    "Return t if PATH contains any excluded directory segment."
    (cl-some (lambda (dir)
               (string-match-p (regexp-quote (concat "/" dir "/")) path))
             exclude-dirs))

  (defun project-todo--valid-extension-p (file extensions)
    "Return t if FILE has one of the allowed EXTENSIONS."
    (cl-some (lambda (ext) (string-suffix-p ext file)) extensions))

  (defun project-todo--collect-files (root extensions exclude-dirs)
    "Return a list of files under ROOT matching EXTENSIONS, skipping EXCLUDE-DIRS."
    (let ((files '()))
      (dolist (f (directory-files-recursively root "." nil))
        (let ((norm (concat "/" (file-relative-name f root))))
          (when (and (project-todo--valid-extension-p f extensions)
                     (not (project-todo--excluded-p norm exclude-dirs)))
            (push f files))))
      (nreverse files)))

  (defun project-todo--scan-file (file root)
    "Return a list of (rel-path line-number comment-text) for every TODO: in FILE."
    (let ((results '())
          (rel (file-relative-name file root)))
      (with-temp-buffer
        (insert-file-contents file)
        (goto-char (point-min))
        (while (re-search-forward "TODO:[[:space:]]*\\(.*\\)" nil t)
          (push (list rel
                      (line-number-at-pos)
                      (string-trim (match-string 1)))
                results)))
      (nreverse results)))

  (defun project-todo--build-content (all-todos)
    "Build the replacement string for the Pending section body."
    (if (null all-todos)
        "   /No TODO comments found in the project./\n"
      (let ((lines (list (format "   /Last scanned: %s — %d TODO(s) across %d file(s)/\n"
                                 (format-time-string "%Y-%m-%d %H:%M")
                                 (apply #'+ (mapcar (lambda (g) (length (cdr g))) all-todos))
                                 (length all-todos)))))
        (dolist (group (nreverse all-todos))
          (let* ((file (car group))
                 (hits (cdr group)))
            (push (format "\n*** ~%s~\n" file) lines)
            (dolist (hit hits)
              (cl-destructuring-bind (rel line text) hit
                (push (format "\n    - [ ] Line %d: %s\n          [[file:%s::%d][→ jump to source]]\n"
                              line
                              (if (string-empty-p text) "/no description/" text)
                              rel
                              line)
                      lines)))))
        (apply #'concat (nreverse lines)))))

  ;; ── main ───────────────────────────────────────────────────────────────────

  (let* ((org-buf (current-buffer))
         (root    (or (bound-and-true-p project-todo-root)
                      (file-name-directory (buffer-file-name))))
         (exts    (or (bound-and-true-p project-todo-extensions)
                      '(".py" ".js" ".ts" ".jsx" ".tsx"
                        ".rb" ".go" ".rs" ".java" ".c" ".cpp" ".h"
                        ".sh" ".bash" ".zsh" ".css" ".scss" ".html"
                        ".el" ".clj" ".ex" ".exs" ".md" ".org"
                        ".tex")))
         (excl    (or (bound-and-true-p project-todo-exclude-dirs)
                      '(".git" "node_modules" ".venv" "venv" "__pycache__"
                        "dist" "build" ".next" "target" "vendor")))
         (files   (project-todo--collect-files root exts excl))
         (all-todos '()))

    ;; Collect all TODOs
    (dolist (file files)
      (let ((hits (project-todo--scan-file file root)))
        (when hits
          (push (cons file hits) all-todos))))

    ;; ── Rewrite the ** Pending section in-buffer ───────────────────────────
    (with-current-buffer org-buf
      (save-excursion
        (goto-char (point-min))
        ;; Find the "** Pending" heading
        (when (re-search-forward "^\\*\\* Pending$" nil t)
          (let ((heading-end (point)))
            ;; Skip past any :PROPERTIES: drawer immediately after the heading
            (forward-line 1)
            (when (looking-at "[[:space:]]*:PROPERTIES:")
              (re-search-forward ":END:" nil t)
              (forward-line 1))
            (let ((content-start (point)))
              ;; Find where this section ends: next same-or-higher heading, or EOF
              (let ((content-end
                     (save-excursion
                       (if (re-search-forward "^\\*\\{1,2\\} " nil t)
                           (line-beginning-position)
                         (point-max)))))
                ;; Replace the section body
                (delete-region content-start content-end)
                (insert "\n" (project-todo--build-content all-todos) "\n"))))))))
  #+END_SRC
